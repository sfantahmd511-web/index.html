import tkinter as tk import random # النص الكامل text = """فريال البطلة، اليوم مو يوم عادي أبدًا، اليوم يوم ميلادچ، اليوم اللي وجودچ بي صار نعمة بحياتي. كل سنة وأنتِ بخير، وكل سنة وأنتِ قريبة مني مثل ما تعودت عليچ، صديقة وأخت بنفس الوقت. أنا ممتنة جدًا لأنچ بحياتي، و لأنچ الشخص اللي أفهمه ويفهمني بدون ما أشرح، واللي أرتاح وياه مهما كان كلشي صعب. صرتِ اليوم بعمر 19، وهذا مو مجرد رقم، هذا بداية مرحلة جديدة وأحلام أكبر. وأني من قلبي أتمنى السنة الجاية أشوفچ احلى طالبة جامعية بالتخصص اللي تحبينه، وتحققين كلشي چنتي تتمني. أريدچ تتذكرين دائمًا إنك قوية، حتى لو شكّيتي بنفسچ مرات، أنتِ گدها وأكثر. وأني راح أبقى دائمًا أشوفچ بطلة، مو بس بكلامي، حتى بكل موقف تثبتين بي نفسچ. أنتِ مو بس صديقتي، أنتِ أختي اللي ما جابتها الأيام بس اختارتها روحي. وأني فخورة بيچ بشكل ما أگدر أوصفه، فخورة بشخصيتچ، بقلبچ، وبكلشي بيچ. وإذا الأيام تعبتچ يوم، تذكري إنو أكو شخص دائمًا واگف وياچ ويدعيلچ من كل گلبة. كل سنة وأنتِ سعادتي، وكل سنة وأنتِ أجمل شيء صار بحياتي. 🎂💖 """ # البيت الشعري poem = """وَكُلُّ صَدِيقٍ لَيْسَ يَصْدُقُ وَعْدَهُ إِلَّا أَنْتِ، كُنْتِ لِلْوُدِّ مِيثَاقَا 💫""" # إنشاء النافذة window = tk.Tk() window.title("🎁 هدية لفريال") window.geometry("700x550") window.config(bg="#8B004B") # شاشة البداية start_frame = tk.Frame(window, bg="#8B004B") start_frame.pack(expand=True) start_label = tk.Label(start_frame, text="🎁 هدية بسيطة لفريال بمناسبة عيد ميلادها الـ 19 💖\nمن المبرمجة سفانة ✨", font=("Arial", 16, "bold"), bg="#8B004B", fg="#FFB6C1", justify="center") start_label.pack(pady=40) def open_card(): start_frame.pack_forget() show_card() open_button = tk.Button(start_frame, text="فتح الهدية 🎀", font=("Arial", 14), bg="#FFB6C1", fg="#8B004B", command=open_card) open_button.pack(pady=20) # عرض البطاقة def show_card(): canvas = tk.Canvas(window, width=700, height=550, bg="#8B004B", highlightthickness=0) canvas.place(x=0, y=0) hearts = [] for i in range(20): x = random.randint(0, 700) y = random.randint(0, 550) size = random.randint(10, 18) heart = canvas.create_text(x, y, text="💖", font=("Arial", size)) hearts.append(heart) def move_hearts(): for heart in hearts: canvas.move(heart, 0, -1) x, y = canvas.coords(heart) if y < 0: canvas.coords(heart, random.randint(0, 700), 550) window.after(50, move_hearts) move_hearts() frame = tk.Frame(window, bg="#8B004B") frame.place(relx=0.5, rely=0.5, anchor="center") decor = tk.Label(frame, text="🎀 💖 🎀 💖 🎀", font=("Arial", 18), bg="#8B004B", fg="#FFB6C1") decor.pack(pady=5) label = tk.Label(frame, text=text, font=("Arial", 11), bg="#8B004B", fg="#FFB6C1", wraplength=600, justify="center") label.pack(pady=5) poem_label = tk.Label(frame, text=poem, font=("Arial", 12, "bold"), bg="#8B004B", fg="#FFC0CB", justify="center") poem_label.pack(pady=10) decor2 = tk.Label(frame, text="💖 🎀 💖 🎀 💖", font=("Arial", 18), bg="#8B004B", fg="#FFB6C1") decor2.pack(pady=5) close_btn = tk.Button(frame, text="سكر البطاقة 💌", command=window.destroy, bg="#FFB6C1", fg="#8B004B", font=("Arial", 12)) close_btn.pack(pady=10) window.mainloop()
